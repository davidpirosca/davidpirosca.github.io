        import java.io.File;
        import javax.sound.sampled.AudioSystem;
        import javax.sound.sampled.Clip;
        
        public class TocaSonidoWav {
        
            public void tocaSonidos(File archivoSonido) {
                try {
                    // Se obtiene un Clip de sonido
                    Clip sonido = AudioSystem.getClip();
        
                    // Se carga con un fichero wav
                    sonido.open(AudioSystem.getAudioInputStream(archivoSonido));
        
                    // Comienza la reproducción
                    sonido.start();
                    Thread.sleep(1);
        
                    // Espera mientras se esté reproduciendo.
                    while (sonido.isRunning()) {
                        Thread.sleep(1);
                    }
                    // Se cierra el clip.
                    sonido.close();
                } catch (Exception e) {
                    System.out.println("" + e);
                }
            }
        }

        import java.io.File;
        import java.util.ArrayList;
        import javax.sound.sampled.AudioFileFormat;
        import javax.sound.sampled.AudioSystem;
        import javax.sound.sampled.Clip;
        
        public class PrincipalSonido {
            final static String RUTA = "sonidos/"; //ruta relativa de sonidos dentro de carpeta de proyecto
        
            public static void main(String arg[]) {
                String cadenasSonidos[] = {"TROMPETA.wav", "ORGANO.wav", "BATERIA.wav"};
                ArrayList ficherosSonidos;
                TocaSonidoWav tocaSonido = new TocaSonidoWav();
                ficherosSonidos = cargaArchivosSonido(cadenasSonidos);
                for (File archivoSonido : ficherosSonidos) {
                    tocaSonido.tocaSonidos(archivoSonido);
                }
            }
        
            private static ArrayList cargaArchivosSonido(String cadenaSonidos[]) {
                ArrayList ficherosSonidos = new ArrayList<>();
                for (String nombre : cadenaSonidos) {
                    ficherosSonidos.add(new File(RUTA + nombre));
                }
                return ficherosSonidos;
            }
        }
