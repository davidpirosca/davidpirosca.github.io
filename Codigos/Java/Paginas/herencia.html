public class Persona {

            String nombre;
            int edad;
            String oculto;
        
            public Persona(String nombre, int edad) {
                this.nombre = nombre;
                this.edad = edad;
                this.oculto = "Pesona";
            }
        
            @Override
            public String toString() {
                return "Persona{" + "nombre=" + nombre + ", edad=" + edad + '}';
            }
        
            String quienSoy() {
                return "SOY UNA PERSONA === " + toString();
            }
        }

        public class Alumno extends Persona{
            String curso;
            String oculto;
        
            public Alumno(String curso, String nombre, int edad) {
                super(nombre, edad);
                this.curso = curso;
                this.oculto = "ALUMNO";
            }
        
            @Override
            public String toString() {
                return super.toString()+"Alumno{" + "curso=" + curso + ", oculto=" + oculto + '}';
            }
            
            public String quienSoy(){
                return "SOY UN ALUMNO ==="+super.quienSoy();
            }
            
        }

        public class Profesor extends Persona {

            String asignatura;
            String nombre;
            String oculto;
        
            public Profesor(String asignatura, String nombre, int edad) {
                super(nombre, edad);
                this.asignatura = asignatura;
                this.nombre = nombre;
                this.oculto = oculto;
            }
        
            @Override
            public String toString() {
                return super.toString() + "Profesor{" + "asignatura=" + asignatura + '}';
            }
        
            public String quienSoy() {
                return "SOY UN PROFESOR === " + super.toString();
            }
        }
