String frase = "Me GUsta merenDAR gaLLEtas";
        String resultado = "";
        char puntero = 0;
        boolean nuevapalabra = true; // Inicializar como true

        for (int i = 0; i < frase.length(); i++) {
            puntero = frase.charAt(i);

            if (frase.charAt(i) == ' ') {
                nuevapalabra = true;
                continue;
            }

            if (nuevapalabra) {
                nuevapalabra = false;
                resultado = resultado + Character.toUpperCase(puntero);
            } else {
                resultado = resultado + Character.toLowerCase(puntero);
            }
        }
        System.out.println("Tu frase seria: " + resultado);
