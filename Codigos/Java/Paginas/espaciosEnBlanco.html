int espaciosEnBlanco = 0;
        
        String frase = "Esta es mi frase de prueba";
        for (int i = 0; i < frase.length(); i++) {
            if (frase.charAt(i) == ' ') {
                espaciosEnBlanco++;
            }
        }
        System.out.println("Numero de espacios en blanco es -> " + espaciosEnBlanco);
